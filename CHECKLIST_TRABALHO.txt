1️⃣  EXTRAIR O ZIP
    $ unzip url_shortener_v2.zip
    $ cd url_shortener_v2

2️⃣  INICIAR COM DOCKER (RECOMENDADO)
    $ docker-compose up --build
    
    ✓ PostgreSQL iniciará em localhost:5432
    ✓ FastAPI iniciará em localhost:5000
    ✓ Docs em http://localhost:5000/docs

3️⃣  TESTAR OS 3 ENDPOINTS OBRIGATÓRIOS

    # Registrar usuário
    curl -X POST "http://localhost:5000/api/v2/auth/register" \
      -H "Content-Type: application/json" \
      -d '{"username":"teste","email":"teste@test.com","password":"senha123"}'

    # Fazer login
    export TOKEN=$(curl -s -X POST "http://localhost:5000/api/v2/auth/login?username=teste&password=senha123" | jq -r '.access_token')

    # Criar uma URL
    curl -X POST "http://localhost:5000/api/v2/urls/" \
      -H "Authorization: Bearer $TOKEN" \
      -H "Content-Type: application/json" \
      -d '{"original_url":"https://google.com"}'

    # ✅ ENDPOINT 2: Listar suas URLs
    curl -X GET "http://localhost:5000/api/v2/urls/user/my-urls" \
      -H "Authorization: Bearer $TOKEN"

    # ✅ ENDPOINT 1: Listar TODAS as URLs (requer admin)
    # (Crie primeiro um usuário admin com is_admin=true)

    # ✅ ENDPOINT 3: Deletar URL
    curl -X DELETE "http://localhost:5000/api/v2/urls/{URL_ID}" \
      -H "Authorization: Bearer $TOKEN"
